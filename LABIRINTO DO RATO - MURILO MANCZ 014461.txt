void layoutLabirinto() {
    char c;
    for (int i = 0; i < LINHAS; i++) {
        for (int j = 0; j < COLUNAS; j++) {
            switch (labirinto[i][j]) {
                case 0:
                    c = ' ';
                    break; //espaço vazio
                case 1:
                    c = '#';
                    break; //paredes
                case 2:
                    c = '@';
                    break;
                case 3:
                    c = ' ';
                    break; //caminho feito pelo rato
                default:
                    c = ' ';
                    break; //diferente é igual a espaço vazio
            }
            cout << c;
        }
        cout << endl;
    }
}

void layoutLabirinto2() {
    char c;
    for (int i = 0; i < LINHAS; i++) {
        for (int j = 0; j < COLUNAS; j++) {
            if (labirinto[i][j] == 0) {
                c = '|';
            }
            if (labirinto[i][j] == 1) {
                c = '|';
            }
            if (labirinto[i][j] == 2) {
                c = 'R';
            }
            if (labirinto[i][j] == 3) {
                c = '#';
            }
            cout << c;
        }
        cout << endl;
    }
}

void AndarLabirinto() {
    system("CLS");
    esquerda = labirinto[l][c-1];
    direita = labirinto[l][c+1];
    cima = labirinto[l-1][c];
    baixo = labirinto[l+1][c];
    if (baixo == 0) {
        l++;
        labirinto[l][c] = 2;
    } else if (direita == 0) {
        c++;
        labirinto[l][c] = 2;
    } else if (cima == 0) {
        l--;
        labirinto[l][c] = 2;
    } else if (esquerda == 0) {
        c--;
        labirinto[l][c] = 2;
    }
    else if (baixo == 2) {
        l++;
        labirinto[l][c] = 2;
        labirinto[l-1][c] = 3;
    } else if (direita == 2) {
        c++;
        labirinto[l][c] = 2;
        labirinto[l][c-1] = 3;
    } else if (cima == 2) {
        l--;
        labirinto[l][c] = 2;
        labirinto[l+1][c] = 3;
    } else if (esquerda == 2) {
        c--;
        labirinto[l][c] = 2;
        labirinto[l][c+1] = 3;
    }
}
